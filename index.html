        items.push(item);
        res.redirect("/");
      }

    app.get("/work", function(req, res){
      res.render("list", {listTitle: "Work list", newListItems: workItems});

    app.post("/work", function(req, res){

      res.redirect("/work");
    });

    app.get("/about", function(req, res){
      res.render("about");
